1、安装vmware tools

2、拷贝VMwareTools-9.9.0-2304977.tar.gz到/tmp 


mount /dev/cdrom /mnt
cp /mnt/VMwareTools-9.9.0-2304977.tar.gz /tmp

3、安装vmware tools 


tar -zxvf VMwareTools-9.9.0-2304977.tar.gz
cd /tmp/vmware-tools-distrib
./vmware-install.pl

4、重启系统 


reboot

5、重新安装vmware tools 


cd /usr/bin
./vmware-config-tools.pl

6、共享文件

7、访问共享文件 


cd /mnt/hgfs
